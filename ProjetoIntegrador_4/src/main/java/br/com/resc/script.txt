CREATE TABLE bem(
	id_bem SERIAL PRIMARY KEY NOT NULL,
	nome VARCHAR(50) NOT NULL,
	custo NUMERIC(8,2) NOT NULL,
	dataAquisicao DATE NOT NULL,
	taxaResidual NUMERIC(4,2) NOT NULL,
	vidaUtil INTEGER NOT NULL,
	turnoDeOito NUMERIC(4,2) NOT NULL,
	tempoDeUso INTEGER NOT NULL,
	id_empresa SERIAL NOT NULL
);

CREATE TABLE empresa(
	id_empresa SERIAL PRIMARY KEY NOT NULL,
	senha VARCHAR(50) NOT NULL,
	cnpj VARCHAR(50) NOT NULL,
	nome VARCHAR(100) NOT NULL,
	areaDeAtuacao VARCHAR(100)
);

ALTER TABLE bem ADD FOREIGN KEY (id_empresa) REFERENCES empresa(id_empresa);
